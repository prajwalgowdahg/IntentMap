```typescript
function extractText(event: Event): string {
  const target = event.target as HTMLElement & { value?: string; textContent?: string | null; innerText?: string; dataset?: DOMStringMap }
  if ('value' in target && typeof target.value === 'string') return target.value
  if (target.dataset?.intent) return target.dataset.intent
  return target.textContent ?? target.innerText ?? ''
}
```
